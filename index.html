        window.calcTotals = (data) => {

            const orders = data.orders || [];

            const extra = data.extraCharge || 0;

            const discInput = data.discount || 1;

            const overrides = data.overrides || {};

            const groups = {};

            orders.forEach(o => {

                if(!groups[o.user]) groups[o.user] = { subtotal: 0, items: [], fullyPaid: true };

                groups[o.user].subtotal += o.total;

                groups[o.user].items.push(o);

                if(!o.paid) groups[o.user].fullyPaid = false;

            });

            const rawOrdersTotal = orders.reduce((a,b)=>a+b.total, 0);

            const userCount = Object.keys(groups).length;

            const totalBeforeDiscount = rawOrdersTotal + extra;

            let totalSavings = 0;

            if (discInput > 0 && discInput <= 1) {

                totalSavings = totalBeforeDiscount - (totalBeforeDiscount * discInput);

            } else {

                totalSavings = Math.abs(discInput === 1 ? 0 : discInput);

            }

            const avgExtra = userCount > 0 ? (extra / userCount) : 0;

            const avgDiscountAmount = userCount > 0 ? (totalSavings / userCount) : 0;

            let overallFinalAdjustedTotal = 0;

            let overallUnpaid = 0;

            Object.entries(groups).forEach(([name, d]) => {

                const calcFinal = d.subtotal + avgExtra - avgDiscountAmount;

                const hasOverride = overrides[name] !== undefined;

                const userFinal = hasOverride ? overrides[name] : Math.max(0, calcFinal);

                groups[name].userFinal = userFinal;

                groups[name].calcFinal = calcFinal;

                groups[name].hasOverride = hasOverride;

                overallFinalAdjustedTotal += userFinal;

                if (!d.fullyPaid) overallUnpaid += userFinal;

            });

            return { groups, overallFinalAdjustedTotal, overallUnpaid };

        };
